CREATE CLASS Order extends V;

CREATE PROPERTY Order.host STRING;
CREATE PROPERTY Order.subTotal DECIMAL;
CREATE PROPERTY Order.shipping DECIMAL;
CREATE PROPERTY Order.tax DECIMAL;
CREATE PROPERTY Order.total DECIMAL;

let productAdmin = create vertex MenuItem set menuItemId = 'productAdmin', label = 'Product Admin', path = '/page/com-networknt-light-v-product-admin-home'
let catalog = create vertex MenuItem set menuItemId = 'catalog', label = 'Catalog', path = '/page/com-networknt-light-v-catalog-home', left = true, roles = 'anonymous, user'
let cart = create vertex MenuItem set menuItemId = 'cart', label = 'Cart', path = '/page/com-networknt-light-v-cart-home', left = false, roles = 'anonymous, user'
let admin = select from MenuItem where menuItemId = 'admin'
create edge Own from $admin to $catalogAdmin
create edge Own from $admin to $productAdmin
